مقدمة في البيانات النقطية

البيانات النقطية (Raster) تمثل البيانات المستمرة في شكل مصفوفة من البكسلات. تستخدم في تمثيل الصور الجوية، الخرائط الطوبوغرافية، بيانات الارتفاع (DEM)، صور الأقمار الصناعية، وغيرها من البيانات المستمرة.

Raster:
مصفوفة من القيم الرقمية تمثل بيانات مستمرة في الفضاء
Pixel (بكسل):
خلية واحدة في مصفوفة البيانات النقطية
Band (نطاق):
قناة واحدة من البيانات (مثل الأحمر، الأخضر، الأزرق في الصور الملونة)

استيراد البيانات النقطية

استخدام raster2pgsql

استيراد ملف Raster
# الصيغة الأساسية
raster2pgsql [options] raster_file [schema.]table | psql [options]

# مثال بسيط
raster2pgsql -s 4326 -I -C -M elevation.tif public.elevation | \
    psql -U postgres -d mydb

# الخيارات الشائعة:
# -s SRID: نظام الإحداثيات
# -I: إنشاء فهرس مكاني
# -C: إضافة قيود
# -M: تحسين الذاكرة
# -F: إضافة عمود اسم الملف
# -t TILE_SIZE: حجم البلاط (مثل 256x256)
# -R: تسجيل كملف خارجي
# -l: إنشاء نظرة عامة (overview)

# مثال مع خيارات متقدمة
raster2pgsql -s 4326 -I -C -M -F -t 256x256 elevation.tif public.elevation | \
    psql -U postgres -d mydb

# استيراد عدة ملفات
raster2pgsql -s 4326 -I -C -M -t 256x256 *.tif public.elevation_tiles | \
    psql -U postgres -d mydb

الاستعلامات على البيانات النقطية

ST_Value - استخراج قيمة

استخراج قيمة من نقطة
-- استخراج قيمة الارتفاع لنقطة معينة
SELECT 
    ST_Value(rast, ST_MakePoint(46.6753, 24.7136)) AS elevation
FROM elevation
WHERE ST_Intersects(rast, ST_MakePoint(46.6753, 24.7136));

-- استخراج قيمة من عدة نقاط
SELECT 
    c.name,
    c.location,
    ST_Value(e.rast, c.location) AS elevation
FROM cities c, elevation e
WHERE ST_Intersects(e.rast, c.location)
ORDER BY elevation DESC;

ST_SummaryStats - الإحصائيات

إحصائيات Raster
-- إحصائيات أساسية
SELECT 
    (ST_SummaryStats(rast)).* 
FROM elevation;

-- إحصائيات مع تفاصيل
SELECT 
    count AS pixel_count,
    sum AS total_value,
    mean AS average_value,
    stddev AS standard_deviation,
    min AS minimum_value,
    max AS maximum_value
FROM (
    SELECT ST_SummaryStats(rast) AS stats FROM elevation
) AS s,
LATERAL (
    SELECT 
        (stats).count,
        (stats).sum,
        (stats).mean,
        (stats).stddev,
        (stats).min,
        (stats).max
) AS stats;

-- إحصائيات لمنطقة معينة
SELECT 
    (ST_SummaryStats(
        ST_Clip(rast, region_boundary)
    )).*
FROM elevation, regions
WHERE regions.name = 'منطقة الرياض'
  AND ST_Intersects(rast, region_boundary);

ST_Clip - قص Raster

قص البيانات النقطية
-- قص Raster بمنطقة معينة
SELECT 
    r.name AS region_name,
    ST_Clip(e.rast, r.boundary) AS clipped_raster
FROM elevation e, regions r
WHERE ST_Intersects(e.rast, r.boundary);

-- حفظ النتائج
CREATE TABLE elevation_regions AS
SELECT 
    r.id,
    r.name,
    ST_Clip(e.rast, r.boundary) AS rast
FROM elevation e, regions r
WHERE ST_Intersects(e.rast, r.boundary);

ST_MapAlgebra - الجبر الخرائطي

عمليات الجبر الخرائطي
-- حساب المنحدر (Slope)
CREATE TABLE slope AS
SELECT 
    ST_Slope(
        rast,
        1,  -- band number
        '32BF',  -- pixel type
        'DEGREES',  -- units
        111120.0  -- scale
    ) AS rast
FROM elevation;

-- حساب الاتجاه (Aspect)
CREATE TABLE aspect AS
SELECT 
    ST_Aspect(
        rast,
        1,
        '32BF',
        'DEGREES',
        111120.0
    ) AS rast
FROM elevation;

-- عمليات حسابية
CREATE TABLE elevation_normalized AS
SELECT 
    ST_MapAlgebra(
        rast,
        1,
        '32BF',
        '[rast] * 0.3048',  -- تحويل من قدم إلى متر
        NULL
    ) AS rast
FROM elevation;

دمج البيانات النقطية والمتجهة

دمج Raster و Vector
-- استخراج قيم Raster للمدن
SELECT 
    c.name,
    c.location,
    ST_Value(e.rast, c.location) AS elevation_meters,
    CASE 
        WHEN ST_Value(e.rast, c.location) < 500 THEN 'منخفض'
        WHEN ST_Value(e.rast, c.location) < 1000 THEN 'متوسط'
        WHEN ST_Value(e.rast, c.location) < 2000 THEN 'مرتفع'
        ELSE 'عالي جداً'
    END AS elevation_category
FROM cities c, elevation e
WHERE ST_Intersects(e.rast, c.location);

-- حساب متوسط الارتفاع لمنطقة
SELECT 
    r.name,
    AVG(ST_Value(e.rast, ST_Centroid(r.boundary))) AS avg_elevation,
    MIN(ST_Value(e.rast, ST_Centroid(r.boundary))) AS min_elevation,
    MAX(ST_Value(e.rast, ST_Centroid(r.boundary))) AS max_elevation
FROM regions r, elevation e
WHERE ST_Intersects(e.rast, r.boundary)
GROUP BY r.id, r.name;

نظام شامل للارتفاع
-- إنشاء قاعدة بيانات للارتفاع
CREATE DATABASE elevation_db;
\c elevation_db
CREATE EXTENSION postgis;
CREATE EXTENSION postgis_raster;

-- استيراد بيانات الارتفاع
-- raster2pgsql -s 4326 -I -C -M -t 256x256 dem.tif public.elevation | psql -U postgres -d elevation_db

-- إنشاء جدول للتحليل
CREATE TABLE elevation_analysis AS
SELECT 
    r.id,
    r.name,
    ST_Clip(e.rast, r.boundary) AS rast,
    (ST_SummaryStats(ST_Clip(e.rast, r.boundary))).mean AS avg_elevation,
    (ST_SummaryStats(ST_Clip(e.rast, r.boundary))).min AS min_elevation,
    (ST_SummaryStats(ST_Clip(e.rast, r.boundary))).max AS max_elevation,
    (ST_SummaryStats(ST_Clip(e.rast, r.boundary))).stddev AS elevation_stddev
FROM regions r, elevation e
WHERE ST_Intersects(e.rast, r.boundary);

-- حساب المنحدر والاتجاه
CREATE TABLE terrain_analysis AS
SELECT 
    id,
    name,
    ST_Slope(rast, 1, '32BF', 'DEGREES', 111120.0) AS slope,
    ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0) AS aspect
FROM elevation_analysis;

-- ربط الارتفاع بالمدن
SELECT 
    c.name,
    c.population,
    ST_Value(e.rast, c.location) AS elevation,
    CASE 
        WHEN ST_Value(e.rast, c.location) < 500 THEN 'سهل'
        WHEN ST_Value(e.rast, c.location) < 1000 THEN 'هضبة'
        ELSE 'جبل'
    END AS terrain_type
FROM cities c, elevation e
WHERE ST_Intersects(e.rast, c.location)
ORDER BY elevation DESC;

التحليل المتقدم للبيانات النقطية

تحليل التضاريس المتقدم

حساب مؤشرات التضاريس
-- حساب مؤشر التضاريس (TRI - Terrain Ruggedness Index)
CREATE FUNCTION calculate_tri(
    rast RASTER,
    band_num INTEGER DEFAULT 1
)
RETURNS RASTER AS $$
BEGIN
    RETURN ST_MapAlgebra(
        ARRAY[rast],
        'tri_calc([rast.val])',
        '32BF',
        'FIRST',
        NULL,
        NULL,
        NULL,
        0,
        0
    );
END;
$$ LANGUAGE plpgsql;

-- حساب مؤشر التضاريس
CREATE TABLE terrain_ruggedness AS
SELECT 
    id,
    ST_MapAlgebra(
        ARRAY[rast],
        'SQRT(POWER([rast.val] - [rast.val:neighborhood], 2))',
        '32BF'
    ) AS tri
FROM elevation;

-- حساب مؤشر التضاريس (TPI - Topographic Position Index)
CREATE TABLE tpi AS
SELECT 
    id,
    ST_MapAlgebra(
        ARRAY[rast],
        '[rast.val] - MEAN([rast.val:neighborhood])',
        '32BF'
    ) AS tpi
FROM elevation;

تحليل المنحدرات المتقدم

تحليل المنحدرات التفصيلي
-- تصنيف المنحدرات
CREATE TABLE slope_classification AS
SELECT 
    id,
    rast,
    CASE 
        WHEN ST_Value(slope_rast, 1, ST_Centroid(ST_Envelope(rast))) < 5 THEN 'مسطح'
        WHEN ST_Value(slope_rast, 1, ST_Centroid(ST_Envelope(rast))) < 15 THEN 'منحدر خفيف'
        WHEN ST_Value(slope_rast, 1, ST_Centroid(ST_Envelope(rast))) < 30 THEN 'منحدر متوسط'
        WHEN ST_Value(slope_rast, 1, ST_Centroid(ST_Envelope(rast))) < 45 THEN 'منحدر شديد'
        ELSE 'منحدر جداً'
    END AS slope_category
FROM (
    SELECT 
        id,
        rast,
        ST_Slope(rast, 1, '32BF', 'DEGREES', 111120.0) AS slope_rast
    FROM elevation
) AS slope_data;

-- حساب اتجاه المنحدر
CREATE TABLE aspect_analysis AS
SELECT 
    id,
    rast,
    ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0) AS aspect_rast,
    CASE 
        WHEN ST_Value(ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0), 
                     1, ST_Centroid(ST_Envelope(rast))) BETWEEN 0 AND 22.5 
             OR ST_Value(ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0), 
                         1, ST_Centroid(ST_Envelope(rast))) BETWEEN 337.5 AND 360 
        THEN 'شمال'
        WHEN ST_Value(ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0), 
                      1, ST_Centroid(ST_Envelope(rast))) BETWEEN 22.5 AND 67.5 
        THEN 'شمال شرق'
        WHEN ST_Value(ST_Aspect(rast, 1, '32BF', 'DEGREES', 111120.0), 
                      1, ST_Centroid(ST_Envelope(rast))) BETWEEN 67.5 AND 112.5 
        THEN 'شرق'
        -- يمكن إضافة باقي الاتجاهات
        ELSE 'غير محدد'
    END AS aspect_direction
FROM elevation;

تحليل الوديان والقمم

تحديد الوديان والقمم
-- تحديد القمم (Peaks)
CREATE TABLE peaks AS
SELECT 
    id,
    ST_Value(rast, 1, point) AS elevation,
    point
FROM (
    SELECT 
        id,
        rast,
        (ST_PixelAsPoints(rast, 1)).geom AS point
    FROM elevation
) AS pixel_points
WHERE ST_Value(rast, 1, point) > (
    SELECT AVG(ST_Value(rast, 1, point)) * 1.5
    FROM elevation, (ST_PixelAsPoints(rast, 1)).geom AS point
);

-- تحديد الوديان (Valleys)
CREATE TABLE valleys AS
SELECT 
    id,
    ST_Value(rast, 1, point) AS elevation,
    point
FROM (
    SELECT 
        id,
        rast,
        (ST_PixelAsPoints(rast, 1)).geom AS point
    FROM elevation
) AS pixel_points
WHERE ST_Value(rast, 1, point) < (
    SELECT AVG(ST_Value(rast, 1, point)) * 0.7
    FROM elevation, (ST_PixelAsPoints(rast, 1)).geom AS point
);

3. نشر البيانات النقطية عبر GeoServer

نشر Raster كخدمة WCS
-- في GeoServer:
-- 1. Stores > Add new store > PostGIS
-- 2. أدخل معلومات الاتصال
-- 3. اختر Coverage (للبيانات النقطية)
-- 4. Publish Coverage
-- 5. حدد:
--    - Native SRS: EPSG:4326
--    - Declared SRS: EPSG:4326
--    - Bounding Boxes: Compute from data
-- 6. اضغط Save

-- إنشاء Coverage View محسّن
CREATE VIEW elevation_coverage AS
SELECT 
    rid,
    ST_Resample(rast, 0.001, 'Bilinear') AS rast  -- تبسيط للعرض
FROM elevation
WHERE ST_Intersects(rast, ST_MakeEnvelope(46.0, 24.0, 50.0, 27.0, 4326));

4. استخدام البيانات النقطية في تطبيقات الويب

إنشاء API للبيانات النقطية
-- دالة لإرجاع بيانات Raster كـ PNG
CREATE OR REPLACE FUNCTION get_raster_image(
    bbox GEOMETRY,
    width INTEGER DEFAULT 256,
    height INTEGER DEFAULT 256
)
RETURNS BYTEA AS $$
DECLARE
    result BYTEA;
BEGIN
    SELECT ST_AsPNG(
        ST_Resize(
            ST_Clip(rast, bbox),
            width,
            height
        )
    )
    INTO result
    FROM elevation
    WHERE ST_Intersects(rast, bbox)
    LIMIT 1;
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;
