مقدمة

الأمان في قواعد البيانات الجغرافية أمر بالغ الأهمية. في هذه المحاضرة سنتعلم أفضل الممارسات لتأمين PostgreSQL و PostGIS، إدارة الصلاحيات، التشفير، والمراقبة.

أمان المصادقة

إدارة كلمات المرور

سياسات كلمات المرور
-- تثبيت pgcrypto للتشفير
CREATE EXTENSION pgcrypto;

-- تشفير كلمة المرور
SELECT crypt('my_password', gen_salt('bf', 10));

-- إنشاء مستخدم بكلمة مرور مشفرة
CREATE USER gis_user WITH 
    PASSWORD 'secure_password_123!'
    VALID UNTIL '2025-12-31';

-- تغيير كلمة المرور
ALTER USER gis_user WITH PASSWORD 'new_secure_password';

-- تعيين انتهاء صلاحية
ALTER USER gis_user VALID UNTIL '2025-12-31';

-- تعطيل مستخدم
ALTER USER gis_user WITH NOLOGIN;

-- تفعيل مستخدم
ALTER USER gis_user WITH LOGIN;

إعدادات المصادقة

تكوين pg_hba.conf
# مثال على إعدادات آمنة في pg_hba.conf

# الاتصال المحلي - peer authentication
local   all             all                                     peer

# الاتصال من localhost - md5
host    all             all             127.0.0.1/32            md5

# الاتصال من شبكة محلية - md5
host    all             all             192.168.1.0/24          md5

# اتصال SSL فقط
hostssl all             all             0.0.0.0/0               md5

# رفض الاتصالات من IP محدد
host    all             all             10.0.0.50/32            reject

# طرق المصادقة:
# trust: بدون مصادقة
# md5: كلمة مرور مشفرة
# scram-sha-256: تشفير قوي
# cert: شهادات SSL
# peer: استخدام نظام التشغيل

إدارة الصلاحيات المتقدمة

الأدوار (Roles)

إنشاء وإدارة الأدوار
-- إنشاء دور
CREATE ROLE gis_readonly;

-- منح صلاحيات للدور
GRANT CONNECT ON DATABASE gis_db TO gis_readonly;
GRANT USAGE ON SCHEMA public TO gis_readonly;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO gis_readonly;

-- إنشاء دور للكتابة
CREATE ROLE gis_writer;
GRANT CONNECT ON DATABASE gis_db TO gis_writer;
GRANT USAGE ON SCHEMA public TO gis_writer;
GRANT SELECT, INSERT, UPDATE ON ALL TABLES IN SCHEMA public TO gis_writer;

-- إنشاء دور للإدارة
CREATE ROLE gis_admin;
GRANT ALL PRIVILEGES ON DATABASE gis_db TO gis_admin;

-- تعيين دور لمستخدم
GRANT gis_readonly TO gis_user;

-- إزالة دور
REVOKE gis_readonly FROM gis_user;

Row Level Security (RLS)

أمان على مستوى الصف
-- تفعيل RLS
ALTER TABLE cities ENABLE ROW LEVEL SECURITY;

-- سياسة للقراءة فقط
CREATE POLICY cities_select_policy ON cities
    FOR SELECT
    TO gis_readonly
    USING (true);

-- سياسة للكتابة
CREATE POLICY cities_insert_policy ON cities
    FOR INSERT
    TO gis_writer
    WITH CHECK (true);

-- سياسة مبنية على المستخدم
CREATE POLICY cities_user_policy ON cities
    FOR ALL
    TO gis_user
    USING (created_by = current_user);

-- عرض السياسات
SELECT * FROM pg_policies WHERE tablename = 'cities';

التشفير

تشفير الاتصالات (SSL)

إعداد SSL
# في postgresql.conf:
ssl = on
ssl_cert_file = 'server.crt'
ssl_key_file = 'server.key'
ssl_ca_file = 'ca.crt'

# التحقق من SSL
SELECT ssl_is_used();

# إجبار SSL في pg_hba.conf
hostssl all all 0.0.0.0/0 md5

تشفير البيانات

تشفير الأعمدة
-- تثبيت pgcrypto
CREATE EXTENSION pgcrypto;

-- تشفير بيانات حساسة
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100),
    password_encrypted BYTEA
);

-- إدراج بيانات مشفرة
INSERT INTO users (username, password_encrypted) VALUES
    ('user1', crypt('password123', gen_salt('bf')));

-- التحقق من كلمة المرور
SELECT username 
FROM users 
WHERE password_encrypted = crypt('password123', password_encrypted);

المراقبة والتدقيق

تفعيل التدقيق (Auditing)

تسجيل العمليات
# في postgresql.conf:
logging_collector = on
log_directory = 'log'
log_filename = 'postgresql-%Y-%m-%d.log'
log_statement = 'all'  # أو 'ddl', 'mod', 'all'
log_connections = on
log_disconnections = on
log_duration = on
log_line_prefix = '%t [%p]: [%l-1] user=%u,db=%d,app=%a,client=%h '

# إعادة تحميل التكوين
SELECT pg_reload_conf();

# عرض السجلات
SELECT * FROM pg_stat_statements
ORDER BY total_time DESC
LIMIT 10;

مراقبة الوصول

مراقبة الجلسات
-- عرض الجلسات النشطة
SELECT 
    pid,
    usename,
    application_name,
    client_addr,
    state,
    query_start,
    query
FROM pg_stat_activity
WHERE state != 'idle'
ORDER BY query_start;

-- إنهاء جلسات مشبوهة
SELECT pg_terminate_backend(pid)
FROM pg_stat_activity
WHERE client_addr = '10.0.0.50'
  AND state != 'idle';

-- مراقبة محاولات الاتصال الفاشلة
SELECT 
    event_time,
    error_severity,
    message
FROM pg_stat_statements
WHERE message LIKE '%authentication%'
ORDER BY event_time DESC;

الأمان على مستوى قاعدة البيانات

قائمة فحص الأمان
-- 1. التحقق من المستخدمين بدون كلمات مرور
SELECT usename FROM pg_user WHERE passwd IS NULL;

-- 2. التحقق من الصلاحيات المفرطة
SELECT 
    grantee,
    table_schema,
    table_name,
    privilege_type
FROM information_schema.role_table_grants
WHERE grantee IN (SELECT rolname FROM pg_roles WHERE rolsuper = true);

-- 3. التحقق من الاتصالات غير المشفرة
SELECT 
    client_addr,
    usename,
    application_name
FROM pg_stat_activity
WHERE ssl = false
  AND client_addr IS NOT NULL;

-- 4. التحقق من الجداول بدون RLS
SELECT 
    schemaname,
    tablename
FROM pg_tables
WHERE schemaname = 'public'
  AND tablename NOT IN (
      SELECT tablename FROM pg_policies
  );
