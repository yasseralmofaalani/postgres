مقدمة

إدارة قواعد البيانات PostgreSQL تتطلب فهمًا عميقًا للعمليات الداخلية، المراقبة، التحسين، والصيانة. في هذه المحاضرة سنتعلم المهارات المتقدمة لإدارة قواعد البيانات الجغرافية بكفاءة.

إدارة قواعد البيانات

إنشاء وإدارة قواعد البيانات

عمليات قواعد البيانات
-- إنشاء قاعدة بيانات
CREATE DATABASE gis_db
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'ar_SA.UTF-8'
    LC_CTYPE = 'ar_SA.UTF-8'
    TEMPLATE = template0;

-- عرض جميع قواعد البيانات
SELECT 
    datname,
    pg_size_pretty(pg_database_size(datname)) AS size,
    pg_database.datcollate AS collation
FROM pg_database
WHERE datistemplate = false
ORDER BY pg_database_size(datname) DESC;

-- عرض حجم قاعدة بيانات
SELECT pg_size_pretty(pg_database_size('gis_db')) AS database_size;

-- عرض معلومات قاعدة بيانات
SELECT 
    datname,
    datdba,
    encoding,
    datcollate,
    datctype
FROM pg_database
WHERE datname = 'gis_db';

-- تغيير مالك قاعدة البيانات
ALTER DATABASE gis_db OWNER TO new_owner;

-- إعادة تسمية قاعدة بيانات
ALTER DATABASE old_name RENAME TO new_name;

-- حذف قاعدة بيانات
DROP DATABASE IF EXISTS gis_db;

إدارة الجداول

معلومات الجداول

استعلام معلومات الجداول
-- عرض جميع الجداول
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) AS total_size,
    pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) AS table_size,
    pg_size_pretty(pg_indexes_size(schemaname||'.'||tablename)) AS indexes_size
FROM pg_tables
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;

-- عدد الصفوف في كل جدول
SELECT 
    schemaname,
    tablename,
    n_live_tup AS row_count
FROM pg_stat_user_tables
ORDER BY n_live_tup DESC;

-- معلومات تفصيلية عن جدول
SELECT 
    column_name,
    data_type,
    character_maximum_length,
    is_nullable,
    column_default
FROM information_schema.columns
WHERE table_schema = 'public'
  AND table_name = 'cities'
ORDER BY ordinal_position;

صيانة الجداول

صيانة الجداول
-- VACUUM: تنظيف وإعادة تنظيم
VACUUM ANALYZE cities;

-- VACUUM كامل
VACUUM FULL cities;

-- VACUUM لجميع الجداول
VACUUM ANALYZE;

-- إعادة بناء الفهارس
REINDEX TABLE cities;

-- إعادة بناء جميع الفهارس
REINDEX DATABASE gis_db;

-- تحديث الإحصائيات
ANALYZE cities;

-- تحديث إحصائيات جميع الجداول
ANALYZE;

المراقبة والأداء

مراقبة الجلسات

مراقبة الجلسات النشطة
-- عرض جميع الجلسات
SELECT 
    pid,
    usename,
    application_name,
    client_addr,
    state,
    query_start,
    state_change,
    wait_event_type,
    wait_event,
    query
FROM pg_stat_activity
WHERE datname = 'gis_db'
  AND state != 'idle'
ORDER BY query_start;

-- إنهاء جلسة
SELECT pg_terminate_backend(pid)
FROM pg_stat_activity
WHERE pid = 12345;

-- إلغاء استعلام
SELECT pg_cancel_backend(pid)
FROM pg_stat_activity
WHERE pid = 12345;

إحصائيات الأداء

إحصائيات الأداء
-- إحصائيات الجداول
SELECT 
    schemaname,
    tablename,
    seq_scan AS sequential_scans,
    seq_tup_read AS seq_tuples_read,
    idx_scan AS index_scans,
    idx_tup_fetch AS index_tuples_fetched,
    n_tup_ins AS inserts,
    n_tup_upd AS updates,
    n_tup_del AS deletes,
    n_live_tup AS live_tuples,
    n_dead_tup AS dead_tuples,
    last_vacuum,
    last_autovacuum,
    last_analyze,
    last_autoanalyze
FROM pg_stat_user_tables
ORDER BY seq_scan DESC;

-- إحصائيات الفهارس
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan AS index_scans,
    idx_tup_read AS tuples_read,
    idx_tup_fetch AS tuples_fetched
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- الاستعلامات البطيئة
SELECT 
    query,
    calls,
    total_time,
    mean_time,
    max_time,
    rows
FROM pg_stat_statements
ORDER BY mean_time DESC
LIMIT 10;

إدارة الاتصالات

إدارة الاتصالات
-- عرض عدد الاتصالات
SELECT 
    count(*) AS total_connections,
    count(*) FILTER (WHERE state = 'active') AS active_connections,
    count(*) FILTER (WHERE state = 'idle') AS idle_connections
FROM pg_stat_activity
WHERE datname = 'gis_db';

-- الحد الأقصى للاتصالات
SHOW max_connections;

-- استخدام الاتصالات
SELECT 
    count(*) AS current_connections,
    (SELECT setting::int FROM pg_settings WHERE name = 'max_connections') AS max_connections,
    round(100.0 * count(*) / 
        (SELECT setting::int FROM pg_settings WHERE name = 'max_connections'), 2) AS usage_percent
FROM pg_stat_activity;

إدارة المستخدمين والصلاحيات

إنشاء وإدارة المستخدمين

إدارة المستخدمين
-- إنشاء مستخدم
CREATE USER gis_user WITH PASSWORD 'secure_password';

-- إنشاء مستخدم مع خيارات
CREATE USER gis_admin WITH 
    PASSWORD 'admin_password'
    CREATEDB
    CREATEROLE
    LOGIN;

-- عرض جميع المستخدمين
SELECT 
    usename,
    usecreatedb,
    usesuper,
    usecanlogin
FROM pg_user
ORDER BY usename;

-- تغيير كلمة المرور
ALTER USER gis_user WITH PASSWORD 'new_password';

-- إعادة تسمية مستخدم
ALTER USER old_name RENAME TO new_name;

-- حذف مستخدم
DROP USER IF EXISTS gis_user;

إدارة الصلاحيات

الصلاحيات
-- منح صلاحيات
GRANT SELECT, INSERT, UPDATE ON cities TO gis_user;
GRANT ALL PRIVILEGES ON DATABASE gis_db TO gis_admin;

-- منح صلاحيات على schema
GRANT USAGE ON SCHEMA public TO gis_user;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO gis_user;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO gis_user;

-- إزالة الصلاحيات
REVOKE INSERT ON cities FROM gis_user;

-- عرض الصلاحيات
SELECT 
    grantee,
    table_schema,
    table_name,
    privilege_type
FROM information_schema.role_table_grants
WHERE table_name = 'cities';

التحسين والصيانة

تحليل الاستعلامات

تحليل الأداء
-- EXPLAIN: خطة التنفيذ
EXPLAIN SELECT * FROM cities WHERE population > 1000000;

-- EXPLAIN ANALYZE: تنفيذ فعلي
EXPLAIN ANALYZE 
SELECT c.name, r.name 
FROM cities c 
JOIN regions r ON ST_Within(c.location, r.boundary);

-- EXPLAIN مع تفاصيل
EXPLAIN (ANALYZE, BUFFERS, VERBOSE, FORMAT JSON)
SELECT * FROM cities WHERE ST_Within(location, 
    (SELECT boundary FROM regions WHERE name = 'منطقة الرياض'));

صيانة دورية

مهام الصيانة
-- سكريبت صيانة شامل
DO $$
DECLARE
    r RECORD;
BEGIN
    -- VACUUM لجميع الجداول
    FOR r IN SELECT tablename FROM pg_tables WHERE schemaname = 'public'
    LOOP
        EXECUTE 'VACUUM ANALYZE ' || quote_ident(r.tablename);
    END LOOP;
    
    -- إعادة بناء الفهارس
    FOR r IN SELECT indexname FROM pg_indexes WHERE schemaname = 'public'
    LOOP
        EXECUTE 'REINDEX INDEX ' || quote_ident(r.indexname);
    END LOOP;
END $$;

-- تحديث الإحصائيات
ANALYZE;
