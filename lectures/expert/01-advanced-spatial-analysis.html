مقدمة

التحليل المكاني المتقدم يتضمن استخدام تقنيات متقدمة لفهم العلاقات المكانية، تحليل الأنماط، وإجراء حسابات معقدة على البيانات الجغرافية. في هذه المحاضرة سنتعلم دوال PostGIS المتقدمة وتقنيات التحليل المكاني الاحترافية.

دوال التحليل المتقدمة

ST_VoronoiPolygons - مخططات فورونوي

مخططات فورونوي تقسم الفضاء إلى مناطق بحيث تكون كل نقطة في منطقة أقرب إلى نقطة معينة من أي نقطة أخرى. مفيدة في تحليل مناطق الخدمة والتأثير.

إنشاء مخططات فورونوي
-- إنشاء مخططات فورونوي للمدن
SELECT 
    ST_VoronoiPolygons(
        ST_Collect(location),
        0,  -- tolerance
        ST_Envelope(ST_Collect(location))  -- extent
    ) AS voronoi
FROM cities;

-- حفظ النتائج في جدول
CREATE TABLE voronoi_diagram AS
SELECT 
    ST_VoronoiPolygons(
        ST_Collect(location),
        0,
        ST_Envelope(ST_Collect(location))
    ) AS geometry
FROM cities;

-- تقسيم المخطط إلى مضلعات منفصلة
SELECT 
    (ST_Dump(
        ST_VoronoiPolygons(
            ST_Collect(location),
            0,
            ST_Envelope(ST_Collect(location))
        )
    )).geom AS voronoi_cell
FROM cities;

ST_ShortestLine - أقصر خط

حساب أقصر خط بين ميزتين جغرافيتين.

حساب أقصر خط
-- أقصر خط بين منطقتين
SELECT 
    r1.name AS region1,
    r2.name AS region2,
    ST_ShortestLine(r1.boundary, r2.boundary) AS shortest_line,
    ST_Length(
        ST_Transform(
            ST_ShortestLine(r1.boundary, r2.boundary),
            3857
        )
    ) / 1000 AS distance_km
FROM regions r1, regions r2
WHERE r1.id < r2.id
  AND ST_DWithin(
      ST_Transform(r1.boundary, 3857),
      ST_Transform(r2.boundary, 3857),
      100000  -- 100 كيلومتر
  )
ORDER BY distance_km;

ST_ClosestPoint - أقرب نقطة

العثور على أقرب نقطة
-- أقرب نقطة على خط من نقطة معينة
SELECT 
    r.name AS road_name,
    c.name AS city_name,
    ST_ClosestPoint(r.geometry, c.location) AS closest_point,
    ST_Distance(
        ST_Transform(c.location, 3857),
        ST_Transform(ST_ClosestPoint(r.geometry, c.location), 3857)
    ) / 1000 AS distance_km
FROM roads r, cities c
ORDER BY distance_km
LIMIT 10;

تحليل التجميع المكاني

تحليل التجميع
-- العثور على المدن المتجمعة
WITH city_clusters AS (
    SELECT 
        c1.id,
        c1.name,
        COUNT(c2.id) AS nearby_cities,
        AVG(ST_Distance(
            ST_Transform(c1.location, 3857),
            ST_Transform(c2.location, 3857)
        )) / 1000 AS avg_distance_km
    FROM cities c1
    LEFT JOIN cities c2 ON 
        ST_DWithin(
            ST_Transform(c1.location, 3857),
            ST_Transform(c2.location, 3857),
            50000  -- 50 كيلومتر
        )
        AND c1.id != c2.id
    GROUP BY c1.id, c1.name, c1.location
)
SELECT *
FROM city_clusters
WHERE nearby_cities >= 3
ORDER BY nearby_cities DESC;

تحليل الاتجاه

تحليل الاتجاه والاتجاهات
-- حساب الاتجاه العام للطرق
SELECT 
    name,
    ST_Azimuth(
        ST_StartPoint(geometry),
        ST_EndPoint(geometry)
    ) * 180 / PI() AS azimuth_degrees,
    CASE 
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 0 AND 45 THEN 'شمال شرق'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 45 AND 90 THEN 'شرق'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 90 AND 135 THEN 'جنوب شرق'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 135 AND 180 THEN 'جنوب'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 180 AND 225 THEN 'جنوب غرب'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 225 AND 270 THEN 'غرب'
        WHEN ST_Azimuth(ST_StartPoint(geometry), ST_EndPoint(geometry)) * 180 / PI() BETWEEN 270 AND 315 THEN 'شمال غرب'
        ELSE 'شمال'
    END AS direction
FROM roads
WHERE ST_GeometryType(geometry) = 'ST_LineString';

CREATE TABLE road_intersections AS
SELECT 
    ST_Intersection(r1.geometry, r2.geometry) AS intersection_point
FROM roads r1, roads r2
WHERE r1.id < r2.id
  AND ST_Intersects(r1.geometry, r2.geometry)
  AND ST_GeometryType(ST_Intersection(r1.geometry, r2.geometry)) = 'ST_Point';

-- حساب طول الشبكة في كل منطقة
SELECT 
    r.name AS region_name,
    COUNT(rd.id) AS road_count,
    SUM(ST_Length(ST_Transform(rd.geometry, 3857))) / 1000 AS total_length_km,
    SUM(ST_Length(ST_Transform(rd.geometry, 3857))) / 1000 / 
        (ST_Area(ST_Transform(r.boundary, 3857)) / 1000000) AS road_density_km_per_sqkm
FROM regions r
LEFT JOIN roads rd ON ST_Intersects(r.boundary, rd.geometry)
GROUP BY r.id, r.name, r.boundary;

تحليل الجوار (Neighborhood Analysis)

تحليل الجوار
-- تحليل الخصائص في الجوار
CREATE FUNCTION analyze_neighborhood(
    center_point GEOMETRY,
    radius_meters NUMERIC
)
RETURNS TABLE(
    city_count INTEGER,
    total_population BIGINT,
    road_length_km NUMERIC,
    avg_distance_km NUMERIC
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        COUNT(DISTINCT c.id)::INTEGER,
        COALESCE(SUM(c.population), 0)::BIGINT,
        COALESCE(SUM(ST_Length(ST_Transform(rd.geometry, 3857))) / 1000, 0),
        COALESCE(AVG(ST_Distance(
            ST_Transform(center_point, 3857),
            ST_Transform(c.location, 3857)
        )) / 1000, 0)
    FROM cities c
    FULL OUTER JOIN roads rd ON ST_DWithin(
        ST_Transform(center_point, 3857),
        ST_Transform(rd.geometry, 3857),
        radius_meters
    )
    WHERE ST_DWithin(
        ST_Transform(center_point, 3857),
        ST_Transform(COALESCE(c.location, rd.geometry), 3857),
        radius_meters
    );
END;
$$ LANGUAGE plpgsql;

-- استخدام الدالة
SELECT * FROM analyze_neighborhood(
    ST_MakePoint(46.6753, 24.7136),
    50000  -- 50 كيلومتر
);

-- إنشاء جدول للتحليل المكاني
CREATE TABLE spatial_analysis_results AS
WITH voronoi AS (
    SELECT ST_VoronoiPolygons(ST_Collect(location), 0) AS geom FROM cities
),
density AS (
    SELECT 
        r.id,
        r.name,
        COUNT(c.id) AS city_count,
        SUM(c.population) AS total_pop,
        ST_Area(ST_Transform(r.boundary, 3857)) / 1000000 AS area_sqkm
    FROM regions r
    LEFT JOIN cities c ON ST_Within(c.location, r.boundary)
    GROUP BY r.id, r.name, r.boundary
),
network AS (
    SELECT 
        r.id,
        COUNT(rd.id) AS road_count,
        SUM(ST_Length(ST_Transform(rd.geometry, 3857))) / 1000 AS road_length_km
    FROM regions r
    LEFT JOIN roads rd ON ST_Intersects(r.boundary, rd.geometry)
    GROUP BY r.id
)
SELECT 
    d.id,
    d.name,
    d.city_count,
    d.total_pop,
    d.area_sqkm,
    d.total_pop / NULLIF(d.area_sqkm, 0) AS population_density,
    n.road_count,
    n.road_length_km,
    n.road_length_km / NULLIF(d.area_sqkm, 0) AS road_density,
    ST_Centroid(d.boundary) AS centroid,
    ST_ConvexHull(d.boundary) AS convex_hull
FROM density d
JOIN network n ON d.id = n.id;

-- عرض النتائج
SELECT 
    name,
    city_count,
    total_pop,
    ROUND(population_density, 2) AS pop_density_per_sqkm,
    ROUND(road_density, 2) AS road_density_km_per_sqkm
FROM spatial_analysis_results
ORDER BY population_density DESC;

التحليل المكاني المتقدم: تقنيات إضافية

تحليل الكثافة النقطية (Point Density Analysis)

تحليل الكثافة النقطية يساعد في تحديد المناطق التي تحتوي على تركيز عالي من النقاط. هذا مفيد في تحليل أنماط الجريمة، مواقع الخدمات، أو أي ظاهرة جغرافية.

حساب الكثافة النقطية
-- إنشاء شبكة للكثافة
CREATE TABLE density_grid AS
WITH grid AS (
    SELECT 
        ST_MakeEnvelope(
            x, y,
            x + 0.01, y + 0.01,
            4326
        ) AS cell
    FROM 
        generate_series(46.0, 50.0, 0.01) AS x,
        generate_series(24.0, 27.0, 0.01) AS y
)
SELECT 
    cell AS geometry,
    COUNT(c.id) AS point_count,
    COUNT(c.id) / ST_Area(ST_Transform(cell, 3857)) * 1000000 AS density_per_sqkm
FROM grid
LEFT JOIN cities c ON ST_Within(c.location, cell)
GROUP BY cell;

-- تحليل الكثافة مع تصنيف
SELECT 
    geometry,
    point_count,
    CASE 
        WHEN point_count = 0 THEN 'منخفضة جداً'
        WHEN point_count BETWEEN 1 AND 5 THEN 'منخفضة'
        WHEN point_count BETWEEN 6 AND 15 THEN 'متوسطة'
        WHEN point_count BETWEEN 16 AND 30 THEN 'عالية'
        ELSE 'عالية جداً'
    END AS density_category
FROM density_grid;

تحليل التجميع المكاني (Spatial Clustering)

تحليل التجميع باستخدام DBSCAN
-- تحديد المجموعات المكانية
CREATE FUNCTION spatial_clusters(
    table_name TEXT,
    geometry_column TEXT,
    eps_distance NUMERIC,
    min_points INTEGER
)
RETURNS TABLE(
    cluster_id INTEGER,
    point_count INTEGER,
    centroid GEOMETRY,
    extent GEOMETRY
) AS $$
DECLARE
    sql_query TEXT;
BEGIN
    -- هذه دالة مبسطة - يمكن تطويرها بشكل أكبر
    sql_query := format('
        WITH points AS (
            SELECT %I FROM %I.%I
        ),
        clusters AS (
            SELECT 
                ROW_NUMBER() OVER () AS cluster_id,
                COUNT(*) AS point_count,
                ST_Centroid(ST_Collect(%I)) AS centroid,
                ST_Envelope(ST_Collect(%I)) AS extent
            FROM points
            GROUP BY cluster_id
        )
        SELECT * FROM clusters
    ', geometry_column, 'public', table_name, geometry_column, geometry_column);
    
    RETURN QUERY EXECUTE sql_query;
END;
$$ LANGUAGE plpgsql;

تحليل الاتجاه والاتجاهات (Directional Analysis)

تحليل الاتجاهات المكانية
-- حساب الاتجاه العام للتوزيع
CREATE FUNCTION calculate_directional_trend(
    points_geometry GEOMETRY
)
RETURNS NUMERIC AS $$
DECLARE
    centroid GEOMETRY;
    max_distance NUMERIC;
    direction_point GEOMETRY;
    azimuth NUMERIC;
BEGIN
    -- حساب النقطة المركزية
    centroid := ST_Centroid(points_geometry);
    
    -- العثور على أبعد نقطة
    SELECT ST_MakePoint(ST_X(location), ST_Y(location))
    INTO direction_point
    FROM cities
    ORDER BY ST_Distance(centroid, location) DESC
    LIMIT 1;
    
    -- حساب الاتجاه
    azimuth := ST_Azimuth(centroid, direction_point) * 180 / PI();
    
    RETURN azimuth;
END;
$$ LANGUAGE plpgsql;

-- استخدام الدالة
SELECT 
    r.name,
    calculate_directional_trend(ST_Collect(c.location)) AS trend_direction
FROM regions r
JOIN cities c ON ST_Within(c.location, r.boundary)
GROUP BY r.id, r.name;

5. الربط مع Leaflet و OpenLayers

Leaflet و OpenLayers هما مكتبات JavaScript لإنشاء خرائط تفاعلية على الويب. يمكن ربطهما مع PostGIS عبر خدمات WMS/WFS أو APIs مخصصة.

إنشاء API للبيانات الجغرافية
-- إنشاء دالة لإرجاع GeoJSON
CREATE OR REPLACE FUNCTION get_cities_geojson(
    bbox GEOMETRY DEFAULT NULL,
    min_population INTEGER DEFAULT 0
)
RETURNS TEXT AS $$
DECLARE
    result TEXT;
BEGIN
    SELECT json_build_object(
        'type', 'FeatureCollection',
        'features', json_agg(
            json_build_object(
                'type', 'Feature',
                'geometry', ST_AsGeoJSON(location)::json,
                'properties', json_build_object(
                    'id', id,
                    'name', name,
                    'population', population
                )
            )
        )
    )::TEXT
    INTO result
    FROM cities
    WHERE population >= min_population
      AND (bbox IS NULL OR ST_Intersects(location, bbox));
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;

-- استخدام الدالة
SELECT get_cities_geojson(

6. الربط مع Python (GeoPandas, Shapely)

استخدام Python للوصول إلى PostGIS
# تثبيت المكتبات المطلوبة
# pip install psycopg2 geopandas sqlalchemy

import psycopg2
import geopandas as gpd
from sqlalchemy import create_engine

# الاتصال بقاعدة البيانات
conn_string = "postgresql://postgres:password@localhost:5432/gis_db"
engine = create_engine(conn_string)

# قراءة البيانات
query = """
    SELECT 
        id,
        name,
        population,
        ST_AsText(location) AS geometry
    FROM cities
    WHERE population > 1000000
"""

# تحميل كـ GeoDataFrame
gdf = gpd.read_postgis(query, engine, geom_col='geometry')

# إجراء تحليل
gdf['area_category'] = gdf['population'].apply(
    lambda x: 'large' if x > 5000000 else 'medium' if x > 2000000 else 'small'
)

# حفظ النتائج مرة أخرى
gdf.to_postgis('cities_analyzed', engine, if_exists='replace', index=False)

# إغلاق الاتصال
engine.dispose()
