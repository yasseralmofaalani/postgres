1. الربط المتقدم مع QGIS

إعداد الاتصال المحسّن

إعدادات متقدمة للاتصال
-- إنشاء مستخدم خاص لـ QGIS
CREATE USER qgis_user WITH PASSWORD 'secure_password';
GRANT CONNECT ON DATABASE gis_db TO qgis_user;
GRANT USAGE ON SCHEMA public TO qgis_user;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO qgis_user;
GRANT SELECT ON ALL SEQUENCES IN SCHEMA public TO qgis_user;

-- إنشاء Views محسّنة للعرض
CREATE VIEW qgis_cities_view AS
SELECT 
    id,
    name,
    population,
    location AS geom,  -- QGIS يتوقع 'geom'
    CASE 
        WHEN population > 5000000 THEN 1
        WHEN population > 2000000 THEN 2
        WHEN population > 1000000 THEN 3
        ELSE 4
    END AS size_code
FROM cities;

-- إضافة عمود للتصنيف
ALTER VIEW qgis_cities_view ADD COLUMN category VARCHAR(50);
UPDATE qgis_cities_view SET category = 
    CASE size_code
        WHEN 1 THEN 'كبيرة جداً'
        WHEN 2 THEN 'كبيرة'
        WHEN 3 THEN 'متوسطة'
        ELSE 'صغيرة'
    END;

2. الربط المتقدم مع ArcGIS

إعداد ArcGIS Enterprise

تكوين ArcGIS Server
-- إنشاء مستخدم لـ ArcGIS
CREATE USER arcgis_user WITH PASSWORD 'secure_password';
GRANT CONNECT ON DATABASE gis_db TO arcgis_user;
GRANT USAGE ON SCHEMA public TO arcgis_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO arcgis_user;

-- إنشاء Feature Class محسّن
CREATE TABLE arcgis_features (
    objectid SERIAL PRIMARY KEY,
    shape GEOMETRY(POINT, 4326),
    name VARCHAR(100),
    population INTEGER,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- إضافة فهرس
CREATE INDEX idx_arcgis_shape ON arcgis_features USING GIST(shape);

-- ArcGIS يتوقع أعمدة معينة
ALTER TABLE arcgis_features 
ADD COLUMN globalid UUID DEFAULT gen_random_uuid();

-- إنشاء View للعرض
CREATE VIEW arcgis_cities_view AS
SELECT 
    id AS objectid,
    location AS shape,
    name,
    population,
    CURRENT_TIMESTAMP AS created_date
FROM cities;

3. نشر الخدمات الجغرافية (GeoServer)

إعداد GeoServer المتقدم

تكوين محسّن
-- إنشاء مستخدم لـ GeoServer
CREATE USER geoserver_user WITH PASSWORD 'secure_password';
GRANT CONNECT ON DATABASE gis_db TO geoserver_user;
GRANT USAGE ON SCHEMA public TO geoserver_user;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO geoserver_user;

-- إنشاء Views محسّنة للخدمات
CREATE VIEW geoserver_cities_wms AS
SELECT 
    id,
    name,
    population,
    ST_Simplify(location, 0.001) AS geometry  -- تبسيط للعرض السريع
FROM cities
WHERE population > 100000;

-- View للـ WFS (تحرير)
CREATE VIEW geoserver_cities_wfs AS
SELECT 
    id,
    name,
    population,
    location AS geometry
FROM cities;

-- <connectionPool>
--     <maxConnections>10</maxConnections>
--     <minConnections>1</minConnections>
--     <connectionTimeout>20</connectionTimeout>
-- </connectionPool>

خدمات WMS, WFS, WCS

نشر الخدمات
-- WMS (Web Map Service) - للعرض فقط
-- في GeoServer:
-- 1. Layers > Add a new resource
-- 2. اختر PostGIS store
-- 3. اختر Table أو View
-- 4. Publish
-- 5. حدد SRS و Bounding Box
-- 6. Save

-- WFS (Web Feature Service) - للتحرير
-- 1. نفس الخطوات
-- 2. في Publishing tab:
--    - Enable WFS
--    - Enable Transactions (للتحرير)
-- 3. Save

-- WCS (Web Coverage Service) - للبيانات النقطية
-- 1. Stores > Add new store > PostGIS
-- 2. اختر Coverage (للـ Raster)
-- 3. Publish Coverage
-- 4. Save

-- استخدام الخدمات في التطبيقات
-- مثال WMS:
-- http://localhost:8080/geoserver/wms?
--     service=WMS&
--     version=1.1.0&
--     request=GetMap&
--     layers=cities&
--     styles=&
--     bbox=46,24,50,27&
--     width=800&
--     height=600&
--     srs=EPSG:4326&
--     format=image/png

-- مثال WFS:
-- http://localhost:8080/geoserver/wfs?
--     service=WFS&
--     version=1.1.0&

Leaflet مع PostGIS

إنشاء API للخرائط
-- دالة لإرجاع GeoJSON
CREATE OR REPLACE FUNCTION get_cities_geojson(
    bbox GEOMETRY DEFAULT NULL,
    min_population INTEGER DEFAULT 0,
    limit_count INTEGER DEFAULT 1000
)
RETURNS TEXT AS $$
DECLARE
    result TEXT;
BEGIN
    SELECT json_build_object(
        'type', 'FeatureCollection',
        'features', json_agg(
            json_build_object(
                'type', 'Feature',
                'id', id,
                'geometry', ST_AsGeoJSON(location)::json,
                'properties', json_build_object(
                    'name', name,
                    'population', population
                )
            )
        )
    )::TEXT
    INTO result
    FROM cities
    WHERE population >= min_population
      AND (bbox IS NULL OR ST_Intersects(location, bbox))
    LIMIT limit_count;
    
    RETURN COALESCE(result, '{"type":"FeatureCollection","features":[]}');
END;
$$ LANGUAGE plpgsql;

OpenLayers مع PostGIS

تكامل OpenLayers
-- دالة لإرجاع Vector Tiles
CREATE OR REPLACE FUNCTION get_vector_tiles(
    z INTEGER,
    x INTEGER,
    y INTEGER
)
RETURNS BYTEA AS $$
DECLARE
    bbox GEOMETRY;
    result BYTEA;
BEGIN
    -- حساب bbox للبلاط
    bbox := ST_TileEnvelope(z, x, y);
    
    -- إرجاع البيانات كـ MVT
    SELECT ST_AsMVT(q, 'cities', 4096, 'geometry')
    INTO result
    FROM (
        SELECT 
            id,
            name,
            population,
            ST_AsMVTGeom(location, bbox, 4096, 256, true) AS geometry
        FROM cities
        WHERE ST_Intersects(location, bbox)
    ) AS q;
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;

5. Python و GeoPandas

تحليل متقدم باستخدام Python
# تثبيت المكتبات
# pip install geopandas psycopg2 sqlalchemy folium

import geopandas as gpd
import pandas as pd
from sqlalchemy import create_engine
import folium

# الاتصال
engine = create_engine('postgresql://postgres:password@localhost:5432/gis_db')

# قراءة البيانات
cities = gpd.read_postgis(
    "SELECT * FROM cities WHERE population > 1000000",
    engine,
    geom_col='location'
)

# إجراء تحليل
cities['density'] = cities['population'] / cities.geometry.buffer(0.01).area

# حفظ النتائج
cities.to_postgis('cities_analyzed', engine, if_exists='replace')

# إنشاء خريطة تفاعلية
m = folium.Map(location=[24.7136, 46.6753], zoom_start=6)
for idx, row in cities.iterrows():
    folium.CircleMarker(
        location=[row.geometry.y, row.geometry.x],
        radius=row['population'] / 100000,
        popup=row['name'],
        color='red',
        fill=True
    ).add_to(m)
m.save('map.html')

6. MapServer

إعداد MapServer
# ملف mapfile.map
MAP
    NAME "PostGIS Map Server"
    STATUS ON
    SIZE 1200 800
    EXTENT 46.0 24.0 50.0 27.0
    UNITS DD
    IMAGECOLOR 255 255 255
    
    # اتصال PostGIS
    CONNECTIONTYPE postgis
    CONNECTION "host=localhost port=5432 dbname=gis_db user=postgres password=pass"
    
    # طبقة المدن
    LAYER
        NAME "cities"
        TYPE POINT
        STATUS ON
        CONNECTION "cities"
        DATA "location FROM cities USING UNIQUE id USING srid=4326"
        
        CLASSITEM "population"
        CLASS
            NAME "Large Cities"
            EXPRESSION ([population] > 5000000)
            STYLE
                SYMBOL "circle"
                SIZE 12
                COLOR 255 0 0
            END
        END
        CLASS
            NAME "Medium Cities"
            EXPRESSION ([population] > 2000000 AND [population] <= 5000000)
            STYLE
                SYMBOL "circle"
                SIZE 8
                COLOR 255 165 0
            END
        END
        CLASS
            NAME "Small Cities"
            EXPRESSION ([population] <= 2000000)
            STYLE
                SYMBOL "circle"
                SIZE 4
                COLOR 0 0 255
            END
        END
    END
    
    # طبقة الطرق
    LAYER
        NAME "roads"
        TYPE LINE
        STATUS ON
        CONNECTION "roads"
        DATA "geometry FROM roads USING UNIQUE id USING srid=4326"
        
        CLASS
            STYLE
                COLOR 128 128 128
                WIDTH 2
            END
        END
    END
END
