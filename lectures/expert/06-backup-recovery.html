مقدمة

النسخ الاحتياطي والاستعادة من أهم عمليات إدارة قواعد البيانات. في هذه المحاضرة سنتعلم أنواع النسخ الاحتياطي المختلفة، كيفية تنفيذها، واستراتيجيات الاستعادة للبيانات الجغرافية.

أنواع النسخ الاحتياطي

1. النسخ الاحتياطي الكامل (Full Backup)

نسخ احتياطي كامل
# نسخ احتياطي SQL نصي
pg_dump -U postgres -d gis_db > backup_full.sql

# نسخ احتياطي مضغوط
pg_dump -U postgres -d gis_db -F c -f backup_full.dump

# نسخ احتياطي مع خيارات
pg_dump -U postgres \
    -d gis_db \
    -F c \
    -f backup_$(date +%Y%m%d_%H%M%S).dump \
    -v \
    --clean \
    --if-exists

# الخيارات:
# -F c: تنسيق مخصص (مضغوط)
# -F p: تنسيق نصي (SQL)
# -F t: تنسيق tar
# -v: تفصيلي
# --clean: إضافة أوامر DROP
# --if-exists: استخدام IF EXISTS مع DROP
# --schema: نسخ schema محدد فقط
# --table: نسخ جدول محدد

2. النسخ الاحتياطي التفاضلي (Differential Backup)

نسخ احتياطي تفاضلي
# نسخ احتياطي للجداول المحددة فقط
pg_dump -U postgres -d gis_db \
    -t cities \
    -t roads \
    -F c \
    -f backup_tables.dump

# نسخ احتياطي لـ schema محدد
pg_dump -U postgres -d gis_db \
    --schema=public \
    -F c \
    -f backup_schema.dump

3. النسخ الاحتياطي المستمر (Continuous Archiving)

إعداد WAL Archiving
# في postgresql.conf:
wal_level = replica
archive_mode = on
archive_command = 'cp %p /backup/wal_archive/%f'

# إنشاء نسخة احتياطي أساسية
pg_basebackup -D /backup/base_backup \
    -Ft \
    -z \
    -P \
    -U postgres

# الخيارات:
# -D: مجلد الوجهة
# -Ft: تنسيق tar
# -z: ضغط
# -P: عرض التقدم

النسخ الاحتياطي المتقدم

نسخ احتياطي مع استبعاد

استبعاد جداول محددة
# نسخ احتياطي مع استبعاد جداول كبيرة
pg_dump -U postgres -d gis_db \
    --exclude-table=large_table \
    --exclude-table-data=temp_table \
    -F c \
    -f backup_excluded.dump

# نسخ احتياطي للهيكل فقط (بدون بيانات)
pg_dump -U postgres -d gis_db \
    --schema-only \
    -F c \
    -f backup_schema_only.dump

# نسخ احتياطي للبيانات فقط (بدون هيكل)
pg_dump -U postgres -d gis_db \
    --data-only \
    -F c \
    -f backup_data_only.dump

نسخ احتياطي متوازي

نسخ احتياطي سريع
# نسخ احتياطي متوازي (للقواعد الكبيرة)
pg_dump -U postgres -d gis_db \
    -F d \
    -j 4 \
    -f backup_directory

# الخيارات:
# -F d: تنسيق directory
# -j: عدد الوظائف المتوازية

الاستعادة

الاستعادة من ملف SQL

استعادة نصية
# استعادة كاملة
psql -U postgres -d gis_db < backup_full.sql

# استعادة مع خيارات
psql -U postgres \
    -d gis_db \
    -f backup_full.sql \
    -v ON_ERROR_STOP=1 \
    -a

# الخيارات:
# -f: ملف SQL
# -v: متغيرات
# -a: عرض جميع الأوامر
# -e: عرض الأوامر المرسلة

الاستعادة من ملف مضغوط

استعادة باستخدام pg_restore
# استعادة كاملة
pg_restore -U postgres -d gis_db backup_full.dump

# استعادة مع خيارات
pg_restore -U postgres \
    -d gis_db \
    -v \
    --clean \
    --if-exists \
    backup_full.dump

# استعادة جدول محدد
pg_restore -U postgres \
    -d gis_db \
    -t cities \
    backup_full.dump

# استعادة schema محدد
pg_restore -U postgres \
    -d gis_db \
    --schema=public \
    backup_full.dump

# استعادة الهيكل فقط
pg_restore -U postgres \
    -d gis_db \
    --schema-only \
    backup_full.dump

# استعادة البيانات فقط
pg_restore -U postgres \
    -d gis_db \
    --data-only \
    backup_full.dump

استراتيجيات النسخ الاحتياطي

النسخ الاحتياطي اليومي

سكريبت نسخ احتياطي يومي
#!/bin/bash
# backup_daily.sh

BACKUP_DIR="/backup/postgresql"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="gis_db"
RETENTION_DAYS=7

# إنشاء النسخة الاحتياطية
pg_dump -U postgres \
    -d $DB_NAME \
    -F c \
    -f $BACKUP_DIR/backup_$DATE.dump \
    -v

# ضغط النسخة الاحتياطية
gzip $BACKUP_DIR/backup_$DATE.dump

# حذف النسخ القديمة
find $BACKUP_DIR -name "backup_*.dump.gz" -mtime +$RETENTION_DAYS -delete

# إرسال إشعار
echo "Backup completed: backup_$DATE.dump.gz" | mail -s "PostgreSQL Backup" admin@example.com

النسخ الاحتياطي التلقائي

إعداد cron job
# إضافة إلى crontab
# crontab -e

# نسخ احتياطي يومي في الساعة 2 صباحاً
0 2 * * * /path/to/backup_daily.sh

# نسخ احتياطي أسبوعي
0 3 * * 0 /path/to/backup_weekly.sh

# نسخ احتياطي شهري
0 4 1 * * /path/to/backup_monthly.sh

التحقق من النسخ الاحتياطي

التحقق من صحة النسخة
# عرض محتويات النسخة الاحتياطية
pg_restore -l backup_full.dump

# التحقق من التكامل
pg_restore --list backup_full.dump | grep -c "TABLE DATA"

# اختبار الاستعادة في قاعدة بيانات تجريبية
createdb test_restore
pg_restore -d test_restore backup_full.dump
# التحقق من البيانات
psql -d test_restore -c "SELECT COUNT(*) FROM cities;"
dropdb test_restore

النسخ الاحتياطي للبيانات الجغرافية

اعتبارات خاصة للبيانات الجغرافية
# نسخ احتياطي مع PostGIS
pg_dump -U postgres -d gis_db \
    --extension=postgis \
    --extension=postgis_topology \
    -F c \
    -f backup_with_postgis.dump

# نسخ احتياطي للبيانات النقطية
pg_dump -U postgres -d gis_db \
    -t elevation \
    -F c \
    -f backup_raster.dump

# التحقق من PostGIS بعد الاستعادة
psql -d restored_db -c "SELECT PostGIS_Version();"
