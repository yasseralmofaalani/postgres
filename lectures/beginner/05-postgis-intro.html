ما هو PostGIS؟

PostGIS هو امتداد مكاني مفتوح المصدر لـ PostgreSQL يضيف دعمًا لأنواع البيانات الجغرافية والوظائف المكانية. يتيح PostGIS تخزين واستعلام البيانات الجغرافية مثل النقاط، الخطوط، المضلعات، والبيانات النقطية (Raster).

معلومة: PostGIS يتبع معايير Open Geospatial Consortium (OGC) وهو متوافق مع معايير SQL/MM للمعلومات المكانية.

مميزات PostGIS

  • دعم أنواع البيانات الجغرافية: Point, LineString, Polygon, MultiPoint, وغيرها
  • مئات الدوال المكانية: للقياس، التحليل، والتحويل
  • الفهارس المكانية: GIST لتحسين أداء الاستعلامات
  • دعم أنظمة الإحداثيات: أكثر من 3000 نظام إحداثي
  • البيانات النقطية: دعم كامل للبيانات النقطية
  • الطوبولوجيا: دعم العلاقات الطوبولوجية

تثبيت PostGIS

التحقق من التثبيت

التحقق من تثبيت PostGIS
-- الاتصال بقاعدة البيانات
psql -U postgres -d mydb

-- التحقق من الإصدار
SELECT PostGIS_Version();

-- عرض معلومات كاملة
SELECT PostGIS_Full_Version();

-- عرض جميع الامتدادات المثبتة
SELECT * FROM pg_extension WHERE extname = 'postgis';

تثبيت PostGIS في قاعدة بيانات

تثبيت PostGIS
-- إنشاء قاعدة بيانات جديدة
CREATE DATABASE gis_db;

-- الاتصال بقاعدة البيانات
\c gis_db

-- تثبيت PostGIS
CREATE EXTENSION postgis;

-- تثبيت امتدادات إضافية (اختياري)
CREATE EXTENSION postgis_topology;    -- للطوبولوجيا
CREATE EXTENSION postgis_raster;      -- للبيانات النقطية
CREATE EXTENSION fuzzystrmatch;       -- للمقارنة النصية الضبابية
CREATE EXTENSION postgis_tiger_geocoder;  -- للجغرافيا الترميزية

أنظمة الإحداثيات (Coordinate Reference Systems)

نظام الإحداثيات يحدد كيفية تمثيل المواقع الجغرافية على سطح الأرض. PostGIS يدعم آلاف أنظمة الإحداثيات المختلفة.

أنظمة الإحداثيات الشائعة

SRID الاسم الوصف الاستخدام
4326 WGS 84 نظام إحداثيات جغرافي عالمي GPS، الخرائط العالمية
3857 Web Mercator إسقاط مركاتور للويب Google Maps، OpenStreetMap
32637 UTM Zone 37N نظام UTM للمنطقة 37 شمال السعودية، الشرق الأوسط
32638 UTM Zone 38N نظام UTM للمنطقة 38 شمال السعودية، الشرق الأوسط
التحقق من أنظمة الإحداثيات
-- عرض معلومات نظام إحداثيات
SELECT * FROM spatial_ref_sys WHERE srid = 4326;

-- البحث عن أنظمة إحداثيات
SELECT srid, auth_name, auth_srid, proj4text 
FROM spatial_ref_sys 
WHERE auth_name = 'EPSG' AND auth_srid = 4326;

-- عرض أنظمة الإحداثيات المتاحة
SELECT srid, auth_name, auth_srid, srtext 
FROM spatial_ref_sys 
WHERE auth_name = 'EPSG' 
ORDER BY auth_srid 
LIMIT 20;

أنواع البيانات الجغرافية الأساسية

1. Point (النقطة)

تمثل موقعًا واحدًا في الفضاء (خط الطول، خط العرض)

إنشاء نقطة
-- إنشاء نقطة باستخدام ST_MakePoint
SELECT ST_MakePoint(46.6753, 24.7136) AS riyadh_point;

-- إنشاء نقطة مع SRID
SELECT ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326) AS riyadh_point;

-- استخدام ST_GeomFromText
SELECT ST_GeomFromText('POINT(46.6753 24.7136)', 4326) AS riyadh_point;

-- استخدام WKT (Well-Known Text)
SELECT 'POINT(46.6753 24.7136)'::geometry AS riyadh_point;

2. LineString (الخط)

تمثل سلسلة من النقاط المتصلة (مثل طريق أو نهر)

إنشاء خط
-- إنشاء خط من عدة نقاط
SELECT ST_MakeLine(
    ST_MakePoint(46.6753, 24.7136),  -- الرياض
    ST_MakePoint(39.1728, 21.4858)   -- جدة
) AS route_line;

-- استخدام WKT
SELECT ST_GeomFromText(
    'LINESTRING(46.6753 24.7136, 39.1728 21.4858)', 
    4326
) AS route_line;

3. Polygon (المضلع)

تمثل منطقة مغلقة (مثل حدود مدينة أو دولة)

إنشاء مضلع
-- إنشاء مضلع بسيط (مربع)
SELECT ST_MakePolygon(
    ST_GeomFromText('LINESTRING(46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7)', 4326)
) AS area_polygon;

-- استخدام WKT
SELECT ST_GeomFromText(
    'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
    4326
) AS area_polygon;

إنشاء جدول مع بيانات جغرافية

إنشاء جدول جغرافي
-- إنشاء جدول للمدن مع موقعها الجغرافي
CREATE TABLE cities (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    population INTEGER,
    location GEOMETRY(POINT, 4326)  -- نقطة في نظام WGS84
);

-- إدراج بيانات
INSERT INTO cities (name, population, location) VALUES
    ('الرياض', 7700000, ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326)),
    ('جدة', 4700000, ST_SetSRID(ST_MakePoint(39.1728, 21.4858), 4326)),
    ('الدمام', 1200000, ST_SetSRID(ST_MakePoint(50.1033, 26.4207), 4326)),
    ('مكة المكرمة', 2400000, ST_SetSRID(ST_MakePoint(39.8579, 21.3891), 4326));

-- عرض البيانات
SELECT 
    name,
    population,
    ST_AsText(location) AS location_wkt,
    ST_X(location) AS longitude,
    ST_Y(location) AS latitude
FROM cities;

الدوال المكانية الأساسية

ST_AsText - تحويل إلى نص

تحويل الهندسة إلى نص
SELECT 
    name,
    ST_AsText(location) AS location_text
FROM cities;

ST_X و ST_Y - استخراج الإحداثيات

استخراج خط الطول والعرض
SELECT 
    name,
    ST_X(location) AS longitude,  -- خط الطول
    ST_Y(location) AS latitude     -- خط العرض
FROM cities;

ST_Distance - حساب المسافة

حساب المسافة بين نقطتين
-- المسافة بين الرياض وجدة (بالدرجات)
SELECT 
    ST_Distance(
        (SELECT location FROM cities WHERE name = 'الرياض'),
        (SELECT location FROM cities WHERE name = 'جدة')
    ) AS distance_degrees;

-- المسافة بالكيلومترات (يجب تحويل إلى نظام إحداثيات مناسب)
SELECT 
    ST_Distance(
        ST_Transform((SELECT location FROM cities WHERE name = 'الرياض'), 3857),
        ST_Transform((SELECT location FROM cities WHERE name = 'جدة'), 3857)
    ) / 1000 AS distance_km;

ST_Area - حساب المساحة

حساب مساحة مضلع
-- إنشاء جدول للمناطق
CREATE TABLE regions (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    boundary GEOMETRY(POLYGON, 4326)
);

-- حساب المساحة (بالمتر المربع بعد التحويل)
SELECT 
    name,
    ST_Area(ST_Transform(boundary, 3857)) AS area_sqm,
    ST_Area(ST_Transform(boundary, 3857)) / 1000000 AS area_sqkm
FROM regions;

التحويل بين أنظمة الإحداثيات

تحويل الإحداثيات
-- تحويل من WGS84 إلى Web Mercator
SELECT 
    name,
    location AS location_wgs84,
    ST_Transform(location, 3857) AS location_mercator
FROM cities;

-- تحويل من WGS84 إلى UTM
SELECT 
    name,
    location AS location_wgs84,
    ST_Transform(location, 32637) AS location_utm
FROM cities;

مثال شامل

مثال شامل: قاعدة بيانات جغرافية
-- إنشاء قاعدة بيانات جغرافية
CREATE DATABASE saudi_arabia_gis;
\c saudi_arabia_gis

-- تثبيت PostGIS
CREATE EXTENSION postgis;

-- جدول المحافظات
CREATE TABLE provinces (
    id SERIAL PRIMARY KEY,
    name_ar VARCHAR(100) NOT NULL,
    name_en VARCHAR(100),
    code VARCHAR(10) UNIQUE,
    population INTEGER,
    area_km2 DECIMAL(10, 2),
    capital_city VARCHAR(100),
    geometry GEOMETRY(POLYGON, 4326),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- جدول المدن
CREATE TABLE cities (
    id SERIAL PRIMARY KEY,
    name_ar VARCHAR(100) NOT NULL,
    name_en VARCHAR(100),
    province_id INTEGER,
    population INTEGER,
    elevation_m INTEGER,
    location GEOMETRY(POINT, 4326),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (province_id) REFERENCES provinces(id)
);

-- جدول الطرق
CREATE TABLE roads (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200),
    road_type VARCHAR(50),  -- highway, main_road, local_road
    length_km DECIMAL(10, 2),
    geometry GEOMETRY(LINESTRING, 4326),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- إدراج بيانات تجريبية
INSERT INTO cities (name_ar, name_en, population, location) VALUES
    ('الرياض', 'Riyadh', 7700000, 
     ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326)),
    ('جدة', 'Jeddah', 4700000, 
     ST_SetSRID(ST_MakePoint(39.1728, 21.4858), 4326));

-- استعلام بسيط
SELECT 
    name_ar,
    population,
    ST_X(location) AS longitude,
    ST_Y(location) AS latitude
FROM cities
ORDER BY population DESC;
