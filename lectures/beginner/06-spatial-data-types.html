نظرة عامة على أنواع البيانات الجغرافية

PostGIS يدعم أنواعًا متعددة من البيانات الجغرافية وفقًا لمعايير OGC. كل نوع مناسب لتمثيل أنواع مختلفة من الميزات الجغرافية.

أنواع البيانات البسيطة

1. POINT (النقطة)

تمثل موقعًا واحدًا في الفضاء ببعدين (X, Y) أو ثلاثة أبعاد (X, Y, Z)

إنشاء واستخدام POINT
-- إنشاء نقطة بسيطة
SELECT ST_MakePoint(46.6753, 24.7136) AS point_2d;

-- إنشاء نقطة ثلاثية الأبعاد
SELECT ST_MakePoint(46.6753, 24.7136, 600) AS point_3d;

-- إنشاء نقطة مع SRID
SELECT ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326) AS riyadh;

-- استخراج الإحداثيات
SELECT 
    ST_X(ST_MakePoint(46.6753, 24.7136)) AS longitude,
    ST_Y(ST_MakePoint(46.6753, 24.7136)) AS latitude;

-- استخدام في جدول
CREATE TABLE landmarks (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    location GEOMETRY(POINT, 4326)
);

2. LINESTRING (الخط)

تمثل سلسلة من النقاط المتصلة لتشكيل خط

إنشاء واستخدام LINESTRING
-- إنشاء خط من نقطتين
SELECT ST_MakeLine(
    ST_MakePoint(46.6753, 24.7136),  -- الرياض
    ST_MakePoint(39.1728, 21.4858)   -- جدة
) AS route;

-- إنشاء خط من عدة نقاط
SELECT ST_MakeLine(ARRAY[
    ST_MakePoint(46.6753, 24.7136),
    ST_MakePoint(46.8000, 24.8000),
    ST_MakePoint(39.1728, 21.4858)
]) AS multi_point_route;

-- استخدام WKT
SELECT ST_GeomFromText(
    'LINESTRING(46.6753 24.7136, 46.8000 24.8000, 39.1728 21.4858)', 
    4326
) AS route;

-- حساب طول الخط
SELECT 
    ST_Length(
        ST_Transform(
            ST_GeomFromText('LINESTRING(46.6753 24.7136, 39.1728 21.4858)', 4326),
            3857
        )
    ) / 1000 AS length_km;

3. POLYGON (المضلع)

تمثل منطقة مغلقة محاطة بحلقة خارجية ويمكن أن تحتوي على ثقوب

إنشاء واستخدام POLYGON
-- إنشاء مضلع بسيط (مربع)
SELECT ST_GeomFromText(
    'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
    4326
) AS square;

-- إنشاء مضلع مع ثقب
SELECT ST_GeomFromText(
    'POLYGON((
        46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7
    ), (
        46.62 24.72, 46.68 24.72, 46.68 24.78, 46.62 24.78, 46.62 24.72
    ))', 
    4326
) AS polygon_with_hole;

-- حساب مساحة المضلع
SELECT 
    ST_Area(
        ST_Transform(
            ST_GeomFromText(
                'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
                4326
            ),
            3857
        )
    ) / 1000000 AS area_sqkm;

أنواع البيانات المركبة

4. MULTIPOINT (نقاط متعددة)

مجموعة من النقاط

إنشاء MULTIPOINT
-- إنشاء نقاط متعددة
SELECT ST_GeomFromText(
    'MULTIPOINT((46.6753 24.7136), (39.1728 21.4858), (50.1033 26.4207))', 
    4326
) AS cities;

-- أو باستخدام ST_Collect
SELECT ST_Collect(ARRAY[
    ST_MakePoint(46.6753, 24.7136),
    ST_MakePoint(39.1728, 21.4858),
    ST_MakePoint(50.1033, 26.4207)
]) AS cities;

5. MULTILINESTRING (خطوط متعددة)

مجموعة من الخطوط

إنشاء MULTILINESTRING
-- إنشاء خطوط متعددة
SELECT ST_GeomFromText(
    'MULTILINESTRING((
        46.6753 24.7136, 46.8000 24.8000
    ), (
        39.1728 21.4858, 39.3000 21.6000
    ))', 
    4326
) AS roads;

6. MULTIPOLYGON (مضلعات متعددة)

مجموعة من المضلعات

إنشاء MULTIPOLYGON
-- إنشاء مضلعات متعددة
SELECT ST_GeomFromText(
    'MULTIPOLYGON((
        ((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))
    ), (
        ((39.1 21.4, 39.2 21.4, 39.2 21.5, 39.1 21.5, 39.1 21.4))
    ))', 
    4326
) AS regions;

7. GEOMETRYCOLLECTION (مجموعة هندسية)

مجموعة من أي أنواع هندسية

إنشاء GEOMETRYCOLLECTION
-- إنشاء مجموعة هندسية
SELECT ST_GeomFromText(
    'GEOMETRYCOLLECTION(
        POINT(46.6753 24.7136),
        LINESTRING(46.6753 24.7136, 39.1728 21.4858),
        POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))
    )', 
    4326
) AS mixed_geometry;

البيانات ثلاثية الأبعاد

البيانات ثلاثية الأبعاد
-- إنشاء نقطة ثلاثية الأبعاد
SELECT ST_MakePoint(46.6753, 24.7136, 600) AS point_3d;

-- إنشاء خط ثلاثي الأبعاد
SELECT ST_GeomFromText(
    'LINESTRING Z (46.6753 24.7136 600, 39.1728 21.4858 50)', 
    4326
) AS line_3d;

-- استخراج الإحداثي Z
SELECT ST_Z(ST_MakePoint(46.6753, 24.7136, 600)) AS elevation;

الدوال للتحقق من نوع البيانات

الدوال للتحقق من النوع
-- التحقق من نوع الهندسة
SELECT 
    ST_GeometryType(ST_MakePoint(46.6753, 24.7136)) AS geom_type;

-- التحقق من البعد
SELECT ST_Dimension(ST_MakePoint(46.6753, 24.7136)) AS dimension;

-- التحقق من عدد الأبعاد الإحداثية
SELECT ST_NDims(ST_MakePoint(46.6753, 24.7136, 600)) AS num_dims;

-- التحقق من نوع معين
SELECT 
    ST_IsValid(ST_MakePoint(46.6753, 24.7136)) AS is_valid,
    ST_IsEmpty(ST_MakePoint(46.6753, 24.7136)) AS is_empty,
    ST_IsSimple(ST_GeomFromText('LINESTRING(0 0, 1 1)')) AS is_simple;

إنشاء جداول بأنواع مختلفة

مثال شامل: جداول بأنواع مختلفة
-- جدول للنقاط (المدن)
CREATE TABLE cities (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    location GEOMETRY(POINT, 4326)
);

-- جدول للخطوط (الطرق)
CREATE TABLE roads (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200),
    road_type VARCHAR(50),
    geometry GEOMETRY(LINESTRING, 4326)
);

-- جدول للمضلعات (المناطق)
CREATE TABLE regions (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    boundary GEOMETRY(POLYGON, 4326)
);

-- جدول لأنواع متعددة (استخدام GEOMETRY العام)
CREATE TABLE features (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    feature_type VARCHAR(50),  -- point, line, polygon
    geometry GEOMETRY,  -- بدون تحديد النوع
    CONSTRAINT check_type CHECK (
        (feature_type = 'point' AND ST_GeometryType(geometry) = 'ST_Point') OR
        (feature_type = 'line' AND ST_GeometryType(geometry) = 'ST_LineString') OR
        (feature_type = 'polygon' AND ST_GeometryType(geometry) = 'ST_Polygon')
    )
);

التحويل بين الأنواع

تحويل الأنواع
-- تحويل MULTIPOINT إلى POINT
SELECT 
    ST_GeometryN(
        ST_GeomFromText('MULTIPOINT((46.6753 24.7136), (39.1728 21.4858))', 4326),
        1
    ) AS first_point;

-- تحويل POLYGON إلى LINESTRING (الحدود الخارجية)
SELECT 
    ST_Boundary(
        ST_GeomFromText(
            'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
            4326
        )
    ) AS boundary_line;

-- تحويل LINESTRING إلى MULTIPOINT
SELECT 
    ST_Collect(
        ST_StartPoint(geom),
        ST_EndPoint(geom)
    ) AS endpoints
FROM (
    SELECT ST_GeomFromText('LINESTRING(46.6753 24.7136, 39.1728 21.4858)', 4326) AS geom
) AS line;

الدوال الهندسية الأساسية

ST_Envelope - المربع المحيط

المربع المحيط
-- الحصول على المربع المحيط
SELECT 
    ST_Envelope(
        ST_GeomFromText(
            'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
            4326
        )
    ) AS bounding_box;

ST_Centroid - النقطة المركزية

النقطة المركزية
-- حساب النقطة المركزية
SELECT 
    ST_Centroid(
        ST_GeomFromText(
            'POLYGON((46.6 24.7, 46.7 24.7, 46.7 24.8, 46.6 24.8, 46.6 24.7))', 
            4326
        )
    ) AS centroid;

ST_Buffer - إنشاء منطقة عازلة

المنطقة العازلة
-- إنشاء منطقة عازلة حول نقطة (بالمتر)
SELECT 
    ST_Buffer(
        ST_Transform(
            ST_MakePoint(46.6753, 24.7136),
            3857
        ),
        5000  -- 5 كيلومتر
    ) AS buffer_zone;

مثال شامل: قاعدة بيانات جغرافية متكاملة

مثال شامل
-- إنشاء قاعدة بيانات
CREATE DATABASE spatial_example;
\c spatial_example
CREATE EXTENSION postgis;

-- جدول المدن (نقاط)
CREATE TABLE cities (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    population INTEGER,
    location GEOMETRY(POINT, 4326)
);

-- جدول الطرق (خطوط)
CREATE TABLE roads (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200),
    road_type VARCHAR(50),
    length_km DECIMAL(10, 2),
    geometry GEOMETRY(LINESTRING, 4326)
);

-- جدول المناطق (مضلعات)
CREATE TABLE administrative_regions (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    code VARCHAR(20),
    area_km2 DECIMAL(12, 2),
    boundary GEOMETRY(POLYGON, 4326)
);

-- إدراج بيانات تجريبية
INSERT INTO cities (name, population, location) VALUES
    ('الرياض', 7700000, ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326)),
    ('جدة', 4700000, ST_SetSRID(ST_MakePoint(39.1728, 21.4858), 4326)),
    ('الدمام', 1200000, ST_SetSRID(ST_MakePoint(50.1033, 26.4207), 4326));

INSERT INTO roads (name, road_type, geometry) VALUES
    ('طريق الرياض-جدة', 'highway',
     ST_GeomFromText('LINESTRING(46.6753 24.7136, 39.1728 21.4858)', 4326));

-- استعلامات
SELECT 
    name,
    ST_AsText(location) AS location,
    ST_X(location) AS longitude,
    ST_Y(location) AS latitude
FROM cities;

SELECT 
    name,
    ST_Length(ST_Transform(geometry, 3857)) / 1000 AS length_km
FROM roads;
