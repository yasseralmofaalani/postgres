الأنواع النصية

الأنواع الرقمية

النوع الوصف النطاق

INTEGER أو INT عدد صحيح -2,147,483,648 إلى 2,147,483,647
BIGINT عدد صحيح كبير -9,223,372,036,854,775,808 إلى 9,223,372,036,854,775,807
DECIMAL(p,s) عدد عشري دقيق حسب الدقة المحددة
NUMERIC(p,s) مثل DECIMAL حسب الدقة المحددة
REAL عدد فاصلة عائمة 6 خانات عشرية
DOUBLE PRECISION عدد فاصلة عائمة مزدوج 15 خانة عشرية
SERIAL عدد تلقائي متزايد 1 إلى 2,147,483,647
BIGSERIAL عدد تلقائي كبير 1 إلى 9,223,372,036,854,775,807

إنشاء جدول مع قيود متعددة

جدول مع قيود
CREATE TABLE employees (
    id SERIAL PRIMARY KEY,

    last_name VARCHAR(50) NOT NULL,

القيود (Constraints)

1. PRIMARY KEY (المفتاح الأساسي)

المفتاح الأساسي
-- طريقة 1: في تعريف العمود
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50)
);

-- طريقة 2: كقيد منفصل
CREATE TABLE users (
    id SERIAL,
    username VARCHAR(50),
    PRIMARY KEY (id)
);

-- مفتاح مركب (Composite Key)
CREATE TABLE order_items (
    order_id INTEGER,
    product_id INTEGER,
    quantity INTEGER,
    PRIMARY KEY (order_id, product_id)
);

2. FOREIGN KEY (المفتاح الخارجي)

المفتاح الخارجي
-- إنشاء الجدول المرجعي أولاً
CREATE TABLE departments (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

-- إنشاء جدول مع مفتاح خارجي
CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    department_id INTEGER,
    FOREIGN KEY (department_id) REFERENCES departments(id)
);

-- مع خيارات ON DELETE و ON UPDATE
CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    department_id INTEGER,
    FOREIGN KEY (department_id) 
        REFERENCES departments(id)
        ON DELETE CASCADE      -- حذف تلقائي عند حذف القسم
        ON UPDATE CASCADE      -- تحديث تلقائي عند تحديث القسم
);

-- خيارات ON DELETE:
-- CASCADE: حذف تلقائي
-- SET NULL: تعيين NULL
-- SET DEFAULT: تعيين القيمة الافتراضية
-- RESTRICT: منع الحذف (الافتراضي)
-- NO ACTION: نفس RESTRICT

3. UNIQUE (فريد)

قيد UNIQUE
-- على عمود واحد
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(100) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE
);

-- على عدة أعمدة
CREATE TABLE user_roles (
    user_id INTEGER,
    role_id INTEGER,
    UNIQUE (user_id, role_id)
);

4. NOT NULL (غير فارغ)

قيد NOT NULL
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    description TEXT,  -- يمكن أن يكون NULL
    stock_quantity INTEGER NOT NULL DEFAULT 0
);

5. CHECK (التحقق)

قيد CHECK
-- قيد على عمود واحد
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price DECIMAL(10, 2) CHECK (price > 0),
    discount_percentage INTEGER CHECK (discount_percentage >= 0 AND discount_percentage <= 100)
);

-- قيد على عدة أعمدة
CREATE TABLE reservations (
    id SERIAL PRIMARY KEY,
    check_in DATE NOT NULL,
    check_out DATE NOT NULL,
    CHECK (check_out > check_in)
);

-- قيد معقد
CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    salary DECIMAL(10, 2),
    bonus DECIMAL(10, 2),
    CHECK (salary + COALESCE(bonus, 0) > 0)
);

6. DEFAULT (القيمة الافتراضية)

القيمة الافتراضية
CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'pending',
    total_amount DECIMAL(10, 2) DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE
);

تعديل الجداول (ALTER TABLE)

إضافة عمود

إضافة عمود
-- إضافة عمود بسيط
ALTER TABLE cities ADD COLUMN country VARCHAR(100);

-- إضافة عمود مع قيود
ALTER TABLE cities 
ADD COLUMN region_id INTEGER 
NOT NULL DEFAULT 1;

-- إضافة عمود مع قيد خارجي
ALTER TABLE cities 
ADD COLUMN region_id INTEGER,
ADD CONSTRAINT fk_region 
FOREIGN KEY (region_id) REFERENCES regions(id);

تعديل عمود

تعديل عمود
-- تغيير نوع البيانات
ALTER TABLE cities ALTER COLUMN population TYPE BIGINT;

-- تغيير القيمة الافتراضية
ALTER TABLE cities ALTER COLUMN is_capital SET DEFAULT FALSE;

-- إضافة قيد NOT NULL
ALTER TABLE cities ALTER COLUMN name SET NOT NULL;

-- إزالة قيد NOT NULL
ALTER TABLE cities ALTER COLUMN description DROP NOT NULL;

-- إعادة تسمية العمود
ALTER TABLE cities RENAME COLUMN name TO city_name;

حذف عمود

حذف عمود
-- حذف عمود
ALTER TABLE cities DROP COLUMN description;

-- حذف عمود مع قيد (CASCADE)
ALTER TABLE cities DROP COLUMN region_id CASCADE;

إضافة وحذف القيود

إدارة القيود
-- إضافة قيد
ALTER TABLE cities 
ADD CONSTRAINT chk_population 
CHECK (population > 0);

-- إضافة مفتاح خارجي
ALTER TABLE cities 
ADD CONSTRAINT fk_region 
FOREIGN KEY (region_id) REFERENCES regions(id);

-- إزالة قيد
ALTER TABLE cities DROP CONSTRAINT chk_population;

-- إعادة تسمية قيد
ALTER TABLE cities RENAME CONSTRAINT old_name TO new_name;

حذف الجداول

حذف الجداول
-- حذف جدول (يجب أن يكون فارغاً أو بدون قيود)
DROP TABLE cities;

-- حذف مع CASCADE (يحذف الجداول المرتبطة)
DROP TABLE regions CASCADE;

-- حذف فقط إذا كان موجوداً
DROP TABLE IF EXISTS cities;

مثال شامل: قاعدة بيانات متكاملة

مثال شامل
-- إنشاء قاعدة بيانات
CREATE DATABASE company_db;

-- الاتصال بقاعدة البيانات
\c company_db

-- جدول الأقسام
CREATE TABLE departments (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    budget DECIMAL(12, 2) CHECK (budget >= 0),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- جدول الموظفين
CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    employee_number VARCHAR(20) UNIQUE NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phone VARCHAR(20),
    salary DECIMAL(10, 2) NOT NULL CHECK (salary > 0),
    department_id INTEGER NOT NULL,
    manager_id INTEGER,
    hire_date DATE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE RESTRICT,
    FOREIGN KEY (manager_id) REFERENCES employees(id) ON DELETE SET NULL,
    
    CONSTRAINT chk_email_format CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$')
);

-- جدول المشاريع
CREATE TABLE projects (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    start_date DATE NOT NULL,
    end_date DATE,
    budget DECIMAL(12, 2) CHECK (budget >= 0),
    status VARCHAR(20) DEFAULT 'planning' CHECK (status IN ('planning', 'active', 'completed', 'cancelled')),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    CHECK (end_date IS NULL OR end_date >= start_date)
);

-- جدول ربط الموظفين بالمشاريع
CREATE TABLE employee_projects (
    employee_id INTEGER NOT NULL,
    project_id INTEGER NOT NULL,
    role VARCHAR(50) NOT NULL,
    hours_allocated INTEGER CHECK (hours_allocated > 0),
    assigned_date DATE DEFAULT CURRENT_DATE,
    
    PRIMARY KEY (employee_id, project_id),
    FOREIGN KEY (employee_id) REFERENCES employees(id) ON DELETE CASCADE,
    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE
);

-- إنشاء فهرس لتحسين الأداء
CREATE INDEX idx_employees_department ON employees(department_id);
CREATE INDEX idx_employees_email ON employees(email);
CREATE INDEX idx_projects_status ON projects(status);

أفضل الممارسات

  • استخدم أسماء واضحة: اختر أسماء جدول وأعمدة وصفية
  • استخدم PRIMARY KEY: كل جدول يجب أن يحتوي على مفتاح أساسي
  • استخدم FOREIGN KEY: للحفاظ على تكامل البيانات
  • استخدم NOT NULL: للأعمدة المطلوبة
  • استخدم CHECK: للتحقق من صحة البيانات
  • استخدم DEFAULT: للقيم الشائعة
  • استخدم INDEX: على الأعمدة المستخدمة في WHERE و JOIN
